Backup Script (Python Example)
import os, shutil, datetime, json

with open("servers.json") as f:
    servers = json.load(f)

for s in servers:
    if s['backup']['enabled']:
        timestamp = datetime.datetime.now().strftime("%Y%m%d-%H%M%S")
        backup_dir = os.path.join(s['backup']['path'], timestamp)
        os.makedirs(backup_dir, exist_ok=True)
        shutil.copytree(s['dir'], backup_dir, dirs_exist_ok=True)
        print(f"Backed up {s['name']} to {backup_dir}")

import os, json, datetime, zipfile, shutil

def backup_server(server):
    timestamp = datetime.datetime.now().strftime("%Y%m%d-%H%M%S")
    backup_file = os.path.join(server['backup_folder'], f"{timestamp}.zip")
    shutil.make_archive(backup_file.replace('.zip',''), 'zip', server['dir'])
    print(f"[{server['name']}] Backup created: {backup_file}")

def restore_server(server, backup_zip):
    shutil.unpack_archive(backup_zip, server['dir'])
    print(f"[{server['name']}] Restored from: {backup_zip}")

void ServerManager::backupConfigs(const ServerConfig& server) {
    QString timestamp = QDateTime::currentDateTime().toString("yyyyMMdd_HHmmss");
    std::string backupFile = server.backup_folder + "/config_" + timestamp.toStdString() + ".zip";
    miniz_cpp::zip_file zip(backupFile);
    zip.add_dir(server.dir + "/Configs");
    zip.save();

    // Keep only last N backups
    cleanupOldBackups(server.backup_folder + "/config_", 5);
}

void ServerManager::backupMap(const ServerConfig& server) {
    QString timestamp = QDateTime::currentDateTime().toString("yyyyMMdd_HHmmss");
    std::string backupFile = server.backup_folder + "/map_" + timestamp.toStdString() + ".zip";
    miniz_cpp::zip_file zip(backupFile);
    zip.add_dir(server.dir + "/Maps");
    zip.save();

    cleanupOldBackups(server.backup_folder + "/map_", 5);
}

void ServerManager::cleanupOldBackups(const std::string& prefix, int keepLastN) {
    // List files matching prefix, sort by timestamp, remove older than N
    std::vector<std::filesystem::path> files;
    for (auto& p : std::filesystem::directory_iterator(std::filesystem::path(prefix).parent_path())) {
        if (p.path().string().find(prefix) != std::string::npos)
            files.push_back(p.path());
    }
    std::sort(files.begin(), files.end(), std::greater<>());
    for (size_t i = keepLastN; i < files.size(); ++i)
        std::filesystem::remove(files[i]);
}

// ---------------------- Config Load/Save ----------------------
ServerManager::ServerManager(const std::string& configFile)
    : configFile(configFile) {}

void ServerManager::loadConfig() {
    std::ifstream file(configFile);
    if (!file.is_open()) return;

    QJsonDocument doc = QJsonDocument::fromJson(QString::fromStdString(
        std::string((std::istreambuf_iterator<char>(file)), std::istreambuf_iterator<char>())
    ).toUtf8());

    if (!doc.isArray()) return;

    servers.clear();
    QJsonArray arr = doc.array();
    for (auto val : arr) {
        QJsonObject obj = val.toObject();
        ServerConfig server;
        server.name = obj["name"].toString().toStdString();
        server.appid = obj["appid"].toInt();
        server.dir = obj["dir"].toString().toStdString();
        server.backup_folder = obj["backup_folder"].toString().toStdString();
        server.auto_update = obj["auto_update"].toBool();
        servers.push_back(server);
    }
}

void ServerManager::saveConfig() {
    QJsonArray arr;
    for (auto& s : servers) {
        QJsonObject obj;
        obj["name"] = QString::fromStdString(s.name);
        obj["appid"] = s.appid;
        obj["dir"] = QString::fromStdString(s.dir);
        obj["backup_folder"] = QString::fromStdString(s.backup_folder);
        obj["auto_update"] = s.auto_update;
        arr.append(obj);
    }
    QJsonDocument doc(arr);
    std::ofstream file(configFile);
    file << doc.toJson().toStdString();
}

QWidget* createSidebar() {
    QWidget* sidebar = new QWidget();
    QVBoxLayout* layout = new QVBoxLayout(sidebar);

    QLineEdit* searchBox = new QLineEdit();
    searchBox->setPlaceholderText("Search servers...");
    layout->addWidget(searchBox);

    QListWidget* serverList = new QListWidget();
    for (auto& s : manager.servers)
        serverList->addItem(QString::fromStdString(s.name));
    layout->addWidget(serverList);

    QPushButton* addBtn = new QPushButton("Add Server");
    layout->addWidget(addBtn);

    QObject::connect(addBtn, &QPushButton::clicked, []() {
        bool ok;
        QString name = QInputDialog::getText(nullptr, "Add Server", "Server Name:", QLineEdit::Normal, "", &ok);
        if (!ok || name.isEmpty()) return;

        int appid = QInputDialog::getInt(nullptr, "Add Server", "Steam AppID:", 0, 0, INT_MAX, 1, &ok);
        if (!ok) return;

        QString dir = QInputDialog::getText(nullptr, "Add Server", "Install Directory:", QLineEdit::Normal, "", &ok);
        if (!ok || dir.isEmpty()) return;

        ServerConfig server;
        server.name = name.toStdString();
        server.appid = appid;
        server.dir = dir.toStdString();
        server.backup_folder = dir.toStdString() + "/Backups";

        manager.servers.push_back(server);
        manager.saveConfig();
        createServerTab(manager.servers.back());
        serverList->addItem(QString::fromStdString(server.name));
    });

    QObject::connect(searchBox, &QLineEdit::textChanged, [=](const QString& text){
        for (int i = 0; i < serverList->count(); ++i) {
            QListWidgetItem* item = serverList->item(i);
            item->setHidden(!item->text().contains(text, Qt::CaseInsensitive));
        }
    });

    QObject::connect(serverList, &QListWidget::itemClicked, [=](QListWidgetItem* item){
        for (int i = 0; i < tabWidget->count(); ++i) {
            if(tabWidget->tabText(i).contains(item->text())){
                tabWidget->setCurrentIndex(i);
                break;
            }
        }
    });

    return sidebar;
}

int main(int argc, char* argv[]) {
    QApplication app(argc, argv);

    manager.loadConfig();

    QMainWindow mainWindow;
    QWidget* central = new QWidget();
    QHBoxLayout* layout = new QHBoxLayout(central);

    QWidget* sidebar = createSidebar();
    layout->addWidget(sidebar);
    layout->addWidget(tabWidget);

    mainWindow.setCentralWidget(central);
    mainWindow.resize(1800, 1000);
    mainWindow.show();

    return app.exec();
}
